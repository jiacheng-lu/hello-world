int removeElement(int* nums, int numsSize, int val){
   int size=0;
   int i=0;
   for(i=0;i<=numsSize-1;i++) {   //从数组nums[]的第一位开始作判断
   if(nums[i]!=val){               //如果数组的第i+1位和给定的值不一样
       nums[size++]=nums[i];        //把第i+1位的值给size+1位，然后size加1
      }
   }             
   return size;              
}
